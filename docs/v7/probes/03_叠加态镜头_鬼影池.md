## 3. 状态机

```
slot: empty --generate--> single --generate--> superposed (n>=2)
superposed --collapse(id)--> collapsed (primary=id)
collapsed --generate--> superposed
any --crystallize--> crystallized (探针10，primary locked)
```

**并发**：同一 slot 生成中禁止第二次 `beginGenerateVideo`（`isGenerating` gate）。

## 4. 数据模型

```typescript
interface ShotCandidate {
  id: string
  shotSlotId: string
  projectId: string
  assetPath: string      // 相对 assets/
  thumbPath: string      // .thumbs/xxx.jpg
  probability: number
  isPrimary: boolean
  promptSnapshot: string
  resonanceHash?: string
  status: 'active' | 'archived'
  createdAt: string
}

interface SuperposedShotState {
  shotSlotId: string
  state: 'empty' | 'single' | 'superposed' | 'collapsed' | 'crystallized'
  primaryId: string | null
  candidateCount: number
}
```

```sql
CREATE TABLE shot_candidates (
  id TEXT PRIMARY KEY,
  shot_slot_id TEXT NOT NULL,
  project_id TEXT NOT NULL,
  asset_path TEXT NOT NULL,
  thumb_path TEXT NOT NULL,
  probability REAL NOT NULL DEFAULT 0.25,
  is_primary INTEGER NOT NULL DEFAULT 0,
  prompt_snapshot TEXT,
  resonance_hash TEXT,
  status TEXT NOT NULL DEFAULT 'active',
  created_at TEXT NOT NULL
);
CREATE INDEX idx_shot_candidates_slot ON shot_candidates(shot_slot_id, status);
```

### shotSlotId 规则

```typescript
export function shotSlotIdForNode(nodeId: string, cellIndex?: number): string {
  if (cellIndex != null) return `storyboard:${nodeId}:${cellIndex}`
  return `videoNode:${nodeId}`
}
```

### collapse 核心逻辑

```typescript
async function collapse(projectId: string, shotSlotId: string, candidateId: string) {
  db.transaction(() => {
    clearPrimary(shotSlotId)
    setPrimary(candidateId)
  })
  const c = getCandidate(candidateId)
  const binding = getBinding(projectId, shotSlotId)
  // 通知 renderer 更新 canvasStore node videoAssetPath
  return { nodeId: binding.nodeId, assetPath: c.assetPath }
}
```
