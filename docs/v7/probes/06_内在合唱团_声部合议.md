## 4. 数据模型

```typescript
type VoiceId = 'impulse' | 'skeptic' | 'nostalgic' | 'perfectionist' | 'bystander' | 'dormant'

interface ChorusVoiceState {
  projectId: string
  volumes: Record<VoiceId, number>
  dominantVoice: VoiceId | null
  updatedAt: string
}

interface ChorusUtterance {
  id: string
  sessionId: string
  voiceId: VoiceId
  text: string
  stance: 'propose' | 'oppose' | 'amend'
  createdAt: string
}

interface ChorusResolution {
  tuningAdjustments: { sourceId?: string; tag?: string; gravityDelta: number }[]
  affectAdjustments?: { timeSec: number; arousalDelta: number }[]
  promptModifiers: string[]
  blockers: string[]
}
```

### LLM 输出 Schema

```json
{
  "utterances": [
    { "voiceId": "skeptic", "text": "...", "stance": "oppose" }
  ],
  "resolution": {
    "tuningAdjustments": [{ "tag": "cyberpunk", "gravityDelta": -0.3 }],
    "promptModifiers": ["desaturate neon"],
    "blockers": []
  }
}
```

### applyResolution

```typescript
export function applyResolution(
  field: ResonanceField,
  res: ChorusResolution,
  envelope?: AffectEnvelope
): { field: ResonanceField; envelope?: AffectEnvelope } {
  let sources = field.sources.map(s => ({ ...s }))
  for (const adj of res.tuningAdjustments) {
    if (adj.sourceId) {
      const i = sources.findIndex(s => s.id === adj.sourceId)
      if (i >= 0) sources[i].gravity = clamp(0, 1, sources[i].gravity + adj.gravityDelta)
    } else if (adj.tag) {
      sources = sources.map(s =>
        s.summary.metaphor.includes(adj.tag) || s.summary.tags?.includes(adj.tag)
          ? { ...s, gravity: clamp(0, 1, s.gravity + adj.gravityDelta) }
          : s
      )
    }
  }
  sources = renormalizeGravity(sources)
  // affect: 同理 arousalSeries 局部 += delta
  return { field: { ...field, sources }, envelope }
}
```
